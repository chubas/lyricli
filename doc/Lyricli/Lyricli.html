def check_params
  self.exit_with_error if @current_track[:artist].nil? or @current_track[:artist].empty?

end

def get_lyrics
  set_current_track
  check_params

  engine = LyricsEngine.new(@current_track[:artist], @current_track[:song])

  begin
    engine.get_lyrics
  rescue LyricsNotFoundException
    "Lyrics not found :("
  end
end
