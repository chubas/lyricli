def load_config
  path = File.expand_path(@config_path)

  if File.exists?(path)
    file = File.new(path, "r")
    @config = MultiJson.decode(file.read)
  else
    load_default_config
  end
end

def save_config
  path = File.expand_path(@config_path)
  file = File.new(path, "w")
  file.print(MultiJson.encode(@config))
  file.close
end
